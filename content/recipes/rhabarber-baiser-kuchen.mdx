ingredients:
  - title: "Für den Teig"
    items:
      - "200 g Weizenmehl"
      - "125 g weiche Butter"
      - "120 g Zucker"
      - "1 Päckchen Vanillezucker"
      - "3 Eigelb"
      - "1 Ei"
      - "2 TL Backpulver"
      - "3 EL Milch"
      - "1 Prise Salz"
  - title: "Für den Belag"
    items:
      - "500 g Rhabarber"
      - "40 g Zucker"
      - "1 EL Speisestärke"
  - title: "Für das Baiser"
    items:
      - "3 Eiweiß"
      - "150 g Zucker"
      - "1 TL Zitronensaft"

steps:
  - title: "Rhabarber vorbereiten"
    description: "Putze den Rhabarber, schneide ihn in Stücke und mische ihn mit Zucker und Stärke, bis alles leicht glänzt."
  - title: "Teig rühren"
    description: "Schlage Butter, Zucker und Vanillezucker cremig, rühre Eigelb und Ei ein und arbeite dann Mehl, Backpulver, Milch und Salz zu einem glatten Teig ein."
  - title: "Form füllen"
    description: "Streiche den Teig in eine gefettete Springform und verteile den Rhabarber darauf, sodass überall etwas Frucht liegt."
  - title: "Kuchen vorbacken"
    description: "Backe den Kuchen, bis der Rand hell goldbraun ist und der Rhabarber sichtbar weich wird."
  - title: "Baiser aufschlagen"
    description: "Schlage das Eiweiß mit Zitronensaft steif und lass den Zucker einrieseln, bis die Masse glänzt und feste Spitzen hält."
  - title: "Haube fertig backen"
    description: "Streiche das Baiser locker auf den Kuchen und backe weiter, bis die Oberfläche trocken wirkt und die Spitzen leicht Farbe bekommen."
  - title: "Abkühlen lassen"
    description: "Lass den Kuchen ruhen, bis die Füllung etwas anzieht und du saftige Stücke mit zarter, leicht knisternder Haube schneiden kannst."
