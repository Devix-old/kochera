ingredients:
  - title: "Für den Salat"
    items:
      - "1 kg festkochende Kartoffeln"
      - "150 g Creme fraîche"
      - "100 g Joghurt natur"
      - "1 kleine Salatgurke"
      - "1 kleine rote Zwiebel"
      - "2 EL Gewürzgurken, fein gewürfelt"
      - "2 EL Gurkenwasser"
      - "1 EL mittelscharfer Senf"
      - "2 EL Schnittlauch, fein geschnitten"
      - "1 EL Dill, fein gehackt"
      - "Salz"
      - "schwarzer Pfeffer"
      - "1 Prise Zucker"

steps:
  - title: "Kartoffeln kochen"
    description: "Koche die Kartoffeln in der Schale weich, lass sie kurz ausdampfen und pelle sie noch warm."
  - title: "Dressing rühren"
    description: "Verrühre Creme fraîche, Joghurt, Senf, Gurkenwasser, Salz, Pfeffer und Zucker zu einer glatten Creme."
  - title: "Gemüse schneiden"
    description: "Schneide Gurke, Gewürzgurken, Zwiebel, Schnittlauch und Dill klein, bis alles frisch und gleichmäßig vorbereitet ist."
  - title: "Kartoffeln mischen"
    description: "Schneide die Kartoffeln in Scheiben und hebe sie mit Gurke, Zwiebel und Kräutern vorsichtig unter das Dressing."
  - title: "Salat ziehen lassen"
    description: "Schmecke alles ab und lass den Salat stehen, bis die Oberfläche cremig wirkt und die Kartoffeln gut Würze aufgenommen haben."
