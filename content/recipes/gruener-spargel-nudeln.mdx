ingredients:
  - title: "Für die Nudeln"
    items:
      - "250 g Spaghetti"
      - "300 g grüner Spargel"
      - "1 Knoblauchzehe"
      - "1 kleine Bio-Zitrone"
      - "2 EL Olivenöl"
      - "100 ml Kochsahne"
      - "40 g Parmesan"
      - "Salz"
      - "Schwarzer Pfeffer"
      - "1 Prise Chiliflocken"
  - title: "Zum Servieren"
    items:
      - "etwas extra Parmesan"
      - "frisch gemahlener Pfeffer"

steps:
  - title: "Wasser erhitzen"
    description: "Setze reichlich Salzwasser auf und bringe es zum Kochen, damit die Pasta später gleichmäßig gar wird."
  - title: "Spargel vorbereiten"
    description: "Wasche den Spargel, schneide die Enden ab und teile die Stangen in mundgerechte Stücke, die Spitzen dürfen etwas länger bleiben."
  - title: "Aromen schneiden"
    description: "Hacke den Knoblauch fein und reibe etwas Zitronenschale ab, damit später alles frisch duftet."
  - title: "Nudeln kochen"
    description: "Koche die Spaghetti bissfest und hebe vor dem Abgießen etwas Kochwasser auf."
  - title: "Spargel anbraten"
    description: "Erhitze das Olivenöl in einer großen Pfanne und brate den Spargel an, bis er leuchtend grün ist und leicht brutzelt."
  - title: "Sauce rühren"
    description: "Gib Knoblauch, Kochsahne, Zitronenschale, Pfeffer und Chiliflocken dazu und rühre alles kurz zusammen."
  - title: "Alles verbinden"
    description: "Mische die Nudeln mit etwas Kochwasser und Parmesan unter, bis die Sauce cremig an den Spaghetti haftet."
  - title: "Abschmecken und servieren"
    description: "Schmecke mit Salz, Zitronensaft und Pfeffer ab und serviere sofort, solange der Spargel noch Biss hat."
