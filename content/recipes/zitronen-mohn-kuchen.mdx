ingredients:
  - title: 'Für den Teig'
    items:
      - '250 g weiche Butter'
      - '200 g Zucker'
      - '1 Päckchen Vanillezucker'
      - '1 Prise Salz'
      - '4 Eier'
      - '2 Bio-Zitronen, Abrieb und 100 ml Saft'
      - '250 g Weizenmehl Type 405'
      - '80 g Mohn'
      - '2 TL Backpulver'
      - '150 g Naturjoghurt'
  - title: 'Für den Guss'
    items:
      - '150 g Puderzucker'
      - '2 bis 3 EL Zitronensaft'
      - '1 TL Mohn'
      - 'etwas Zitronenabrieb nach Wunsch'

steps:
  - title: 'Form vorbereiten'
    description: 'Fette eine Kastenform sorgfältig ein und heize den Ofen vor, damit der Teig gleichmäßig aufgehen kann.'
  - title: 'Butter cremig rühren'
    description: 'Rühre Butter, Zucker, Vanillezucker und Salz, bis die Masse hell aussieht und weich vom Rand löst.'
  - title: 'Eier einarbeiten'
    description: 'Gib die Eier nacheinander dazu und rühre nur so lange, bis alles glatt und cremig verbunden ist.'
  - title: 'Trockene Zutaten mischen'
    description: 'Vermische Mehl, Mohn und Backpulver gründlich, damit sich der Mohn später ruhig in der Krume verteilt.'
  - title: 'Teig vollenden'
    description: 'Rühre Mehlmischung, Joghurt, Zitronensaft und Zitronenabrieb kurz unter, bis ein zäher, frischer Teig entsteht.'
  - title: 'Teig einfüllen'
    description: 'Streiche den Teig in die Form, glätte die Oberfläche locker und ziehe mittig eine kleine Spur für einen schönen Aufriss.'
  - title: 'Kuchen backen'
    description: 'Backe den Kuchen, bis er goldgelb duftet, die Oberfläche leicht aufreißt und ein Holzstäbchen sauber herauskommt.'
  - title: 'Guss rühren'
    description: 'Verrühre Puderzucker mit Zitronensaft zu einem dickflüssigen Guss, der langsam vom Löffel läuft.'
  - title: 'Abkühlen und glasieren'
    description: 'Lass den Kuchen vollständig abkühlen und verteile den Guss ungleichmäßig darüber, damit er sich fein an die Oberfläche setzt.'
