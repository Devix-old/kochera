ingredients:
  - title: 'Für den Mürbeteigboden'
    items:
      - '250 g Weizenmehl'
      - '125 g kalte Butter'
      - '80 g Zucker'
      - '1 Ei'
      - '1 Prise Salz'
  - title: 'Für den Vanillepudding'
    items:
      - '500 ml Milch'
      - '1 Päckchen Vanillepuddingpulver'
      - '40 g Zucker'
      - '1 TL Vanilleextrakt'
  - title: 'Für den Belag'
    items:
      - '500 g Erdbeeren'
      - '1 Päckchen klarer Tortenguss'
      - '250 ml Wasser'
      - '2 EL Zucker'
      - '2 EL Aprikosenmarmelade optional'

steps:
  - title: 'Teig kneten'
    description: 'Verknete Mehl, Butter, Zucker, Ei und Salz rasch zu einem glatten Teig und forme ihn flach.'
  - title: 'Teig kühlen'
    description: 'Lege den Teig kalt, damit er fest wird und sich später sauber in die Form drücken lässt.'
  - title: 'Boden auslegen'
    description: 'Drücke den Teig in eine gefettete Form, ziehe einen kleinen Rand hoch und stich den Boden mehrmals ein.'
  - title: 'Boden backen'
    description: 'Backe den Boden goldgelb, bis er trocken wirkt und am Rand leicht Farbe bekommt.'
  - title: 'Pudding kochen'
    description: 'Koche aus Milch, Puddingpulver, Zucker und Vanille einen dicken Pudding und decke ihn direkt an der Oberfläche ab.'
  - title: 'Pudding aufstreichen'
    description: 'Streiche den abgekühlten Pudding glatt auf den kalten Boden, damit später nichts verrutscht.'
  - title: 'Erdbeeren vorbereiten'
    description: 'Wasche die Erdbeeren, tupfe sie gut trocken und halbiere größere Früchte.'
  - title: 'Kuchen belegen'
    description: 'Setze die Erdbeeren dicht auf den Pudding, bis kaum noch Lücken zu sehen sind.'
  - title: 'Guss verteilen'
    description: 'Rühre den Guss an und gib ihn gleichmäßig über die Erdbeeren, bis alles leicht glänzt.'
