ingredients:
  - title: "Für den Boden"
    items:
      - "250 g Mehl"
      - "125 g kalte Butter"
      - "80 g Zucker"
      - "1 Ei"
      - "1 Prise Salz"
      - "1 TL Backpulver"
  - title: "Für die Fruchtschicht"
    items:
      - "500 g Stachelbeeren"
      - "60 g Zucker"
      - "1 EL Speisestärke"
      - "1 TL Zitronensaft"
  - title: "Für das Baiser"
    items:
      - "3 Eiweiß"
      - "150 g Zucker"
      - "1 TL Vanillezucker"
      - "1 TL Speisestärke"

steps:
  - title: "Teig kneten"
    description: "Verknete Mehl, Butter, Zucker, Ei, Salz und Backpulver zügig zu einem glatten Teig und drücke ihn in die Form, bis ein kleiner Rand steht."
  - title: "Boden vorbereiten"
    description: "Stich den Boden mehrmals ein und backe ihn kurz an, damit er später unter den Früchten stabil bleibt."
  - title: "Früchte mischen"
    description: "Vermenge die Stachelbeeren mit Zucker, Stärke und Zitronensaft, bis alles leicht glänzt und die Stärke gleichmäßig haftet."
  - title: "Eiweiß aufschlagen"
    description: "Schlage das Eiweiß an und lass den Zucker nach und nach einrieseln, bis die Masse dick, glänzend und standfest ist."
  - title: "Kuchen schichten"
    description: "Verteile zuerst die Stachelbeeren auf dem Boden und streiche dann das Baiser locker darüber, damit kleine Spitzen stehen bleiben."
  - title: "Golden backen"
    description: "Backe den Kuchen zu Ende, bis die Haube hell goldgelb aussieht, fein duftet und beim Antippen zart fest wirkt."
  - title: "Abkühlen lassen"
    description: "Lass den Kuchen vollständig ruhen, damit sich Saft, Boden und Haube setzen und die Stücke sauber geschnitten werden können."
