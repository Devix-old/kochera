ingredients:
  - title: "Smeten"
    items:
      - "3,5 dl vetemjöl"
      - "2 msk strösocker"
      - "2 msk bakpulver"
      - "0,5 tsk salt"
      - "1 tsk vaniljsocker"
      - "3 dl havremjölk"
      - "1 msk äppelcidervinäger"
      - "3 msk smält neutral olja eller smör"
  - title: "Till servering"
    items:
      - "1 dl lönnsirap"
      - "2 dl färska bär"
      - "1 klick smör"
steps:
  - title: "Förbered vätskan"
    description: "Blanda mjölk och äppelcidervinäger i en kanna och låt det stå i cirka 5–10 minuter för att reagera."
  - title: "Blanda torrt"
    description: "Vispa ihop vetemjöl, socker, bakpulver, salt och vaniljsocker i en stor bunke så att allt blandas ordentligt."
  - title: "Tillsätt vätskan"
    description: "Häll ner mjölkblandningen och det smälta fettet i de torra ingredienserna och rör om försiktigt med en visp."
  - title: "Vila smeten"
    description: "Låt pannkakssmeten vila på köksbänken i minst 10 minuter; detta är hemligheten bakom den perfekta fluffigheten."
  - title: "Stek i pannan"
    description: "Hetta upp en panna på medelvärme och klicka ut små cirklar. Vänd när du ser små bubblor på ovansidan."
  - title: "Servera direkt"
    description: "Stapla pannkakorna på en tallrik och toppa med rikligt med sirap, bär och en liten klick smör."
