ingredients:
  - title: "Für den Boden"
    items:
      - "200 g Mehl"
      - "120 g Zucker"
      - "1 Päckchen Vanillezucker"
      - "3 Eier"
      - "100 g weiche Butter"
      - "1 TL Backpulver"
      - "1 Prise Salz"
      - "3 EL Milch"
  - title: "Für die Creme"
    items:
      - "1 Päckchen Vanillepuddingpulver"
      - "400 ml Milch"
      - "40 g Zucker"
      - "250 g Magerquark"
      - "200 ml Schlagsahne"
  - title: "Für den Belag"
    items:
      - "300 g Rhabarber"
      - "250 g Erdbeeren"
      - "2 EL Zucker"
      - "1 EL Zitronensaft"
      - "1 Päckchen klarer Tortenguss"
      - "250 ml Wasser"
      - "2 EL gehobelte Mandeln"

steps:
  - title: "Heize den Ofen vor"
    description: "Lege den Boden einer Springform aus, fette den Rand leicht und heize den Ofen vor, damit der Teig später gleichmäßig aufgeht."
  - title: "Rühre den Boden"
    description: "Schlage Butter, Zucker und Eier glatt, rühre Mehl, Backpulver, Salz und Milch ein und verteile den Teig in der Form."
  - title: "Backe den Boden"
    description: "Backe den Boden, bis er goldgelb aussieht und warm nach Butter duftet, dann lass ihn vollständig auskühlen."
  - title: "Koche die Creme"
    description: "Koche aus Puddingpulver, Milch und Zucker eine dicke Creme, lass sie abkühlen und hebe dann Quark und geschlagene Sahne unter."
  - title: "Bereite den Belag zu"
    description: "Gare den Rhabarber mit Zucker und Zitronensaft kurz, gib die Erdbeeren dazu und binde alles mit dem Tortenguss, bis die Masse leicht glänzt."
  - title: "Setze die Torte zusammen"
    description: "Streiche die Creme auf den kalten Boden, verteile den Fruchtbelag darauf und streue zum Schluss die kurz gerösteten Mandeln über die Torte."
