ingredients:
  - title: "Für den Spargel"
    items:
      - "1,2 kg weißer Spargel"
      - "40 g Butter oder laktosefreie Butter"
      - "1 EL Zitronensaft"
      - "1 TL Zucker"
      - "1 TL Salz"
      - "2 EL Wasser"
      - "2 Zitronenscheiben"
  - title: "Zum Servieren"
    items:
      - "1 EL gehackte Petersilie"
      - "etwas schwarzer Pfeffer"

steps:
  - title: "Spargel vorbereiten"
    description: "Schäle den Spargel gründlich, schneide die holzigen Enden ab und heize den Ofen vor, damit später alles gleichmäßig gart."
  - title: "Folie belegen"
    description: "Lege den Spargel auf ein großes Stück Alufolie, gib Butter, Zitronensaft, Zucker, Salz, Wasser und Zitronenscheiben darauf und verteile alles locker."
  - title: "Paket verschließen"
    description: "Falte die Folie dicht zu, damit der Dampf im Paket bleibt und der Spargel innen weich wird statt außen trocken."
  - title: "Im Ofen garen"
    description: "Backe das Paket, bis die Stangen weich sind und beim Einstechen nur noch leicht Widerstand haben."
  - title: "Servieren"
    description: "Öffne die Folie vorsichtig, streue Petersilie und etwas Pfeffer darüber und serviere den Spargel sofort, solange er noch glänzt und warm ist."
