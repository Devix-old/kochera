ingredients:
  - title: "Für Patties & Buns"
    items:
      - "600 g Rinderhack, am besten mit etwas Fett"
      - "4 Burger-Buns"
      - "1 TL Salz"
      - "1/2 TL schwarzer Pfeffer"
      - "1 TL mittelscharfer Senf"
      - "1 EL neutrales Öl"
      - "1 EL Butter"
  - title: "Für Sauce & Belag"
    items:
      - "4 EL Mayonnaise"
      - "2 EL Ketchup"
      - "1 TL Senf"
      - "1 große Tomate"
      - "1 kleine rote Zwiebel"
      - "4 Blätter Romana-Salat"
      - "8 Gewürzgurkenscheiben"

steps:
  - title: "Rühre die Sauce"
    description: "Verrühre Mayonnaise, Ketchup und Senf zu einer glatten Sauce."
  - title: "Schneide den Belag"
    description: "Schneide Tomate und Zwiebel in dünne Scheiben und trockne den Salat gut ab."
  - title: "Forme die Patties"
    description: "Mische das Hack nur kurz mit Salz, Pfeffer und Senf und forme vier lockere Patties mit einer kleinen Mulde in der Mitte."
  - title: "Röste die Buns"
    description: "Streiche die Schnittflächen dünn mit Butter ein und röste sie an, bis sie goldbraun sind und warm duften."
  - title: "Brate das Fleisch"
    description: "Brate die Patties in heißem Öl, bis es kräftig brutzelt und außen eine dunkle Kruste entsteht."
  - title: "Baue die Burger"
    description: "Bestreiche die Buns mit Sauce und schichte Salat, Patty, Tomate, Gurke und Zwiebel darauf."
  - title: "Setze den Deckel auf"
    description: "Lege die oberen Bun-Hälften auf und drücke die Burger nur leicht zusammen, damit sie saftig bleiben."
