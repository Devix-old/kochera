ingredients:
  - title: "Für die Creme"
    items:
      - "250 g Mascarpone"
      - "250 g Magerquark"
      - "200 g Schlagsahne"
      - "60 g Zucker"
      - "1 Päckchen Vanillezucker"
      - "1 TL Zitronenabrieb"
  - title: "Für die Frucht- und Keks-Schichten"
    items:
      - "300 g Himbeeren"
      - "180 g Spekulatius"
      - "2 EL Himbeerkonfitüre"
      - "1 EL Zitronensaft"
  - title: "Für oben drauf"
    items:
      - "80 g Himbeeren"
      - "2 Spekulatius-Kekse, grob zerbröselt"

steps:
  - title: "Creme anrühren"
    description: "Rühre Mascarpone, Quark, Zucker, Vanillezucker und Zitronenabrieb glatt und hebe die steif geschlagene Sahne locker unter, bis die Creme dick und weich wirkt."
  - title: "Himbeeren vorbereiten"
    description: "Mische die Himbeeren mit Zitronensaft und glatt gerührter Konfitüre, bis sie glänzen und leicht Saft ziehen."
  - title: "Spekulatius einlegen"
    description: "Verteile eine erste Lage grob gebrochene Spekulatius in der Form, damit der Boden locker bedeckt ist."
  - title: "Schichten aufbauen"
    description: "Gib etwas Creme auf die Kekse, verteile dann Himbeeren darüber und schichte alles noch einmal, bis die Form gefüllt ist."
  - title: "Kalt stellen"
    description: "Stelle das Dessert für 25 Minuten kalt, bis die Creme anzieht und die Schichten beim Anschneiden ruhig bleiben."
  - title: "Fertig servieren"
    description: "Bestreue die Oberfläche mit Keksbröseln und frischen Himbeeren, wenn die Creme kühl, fest und schön cremig ist."
