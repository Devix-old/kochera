ingredients:
  - title: "Für den Boden"
    items:
      - "250 g Mehl"
      - "125 g kalte Butter"
      - "80 g Zucker"
      - "1 Ei"
      - "1 Prise Salz"
      - "1 TL Backpulver"
  - title: "Für die Quarkfüllung"
    items:
      - "500 g Quark"
      - "200 g Schmand"
      - "120 g Zucker"
      - "2 Eier"
      - "1 Päckchen Vanillezucker"
      - "1 Päckchen Vanillepuddingpulver"
      - "50 ml neutrales Öl"
      - "abgeriebene Schale von 1 Bio-Zitrone"
  - title: "Für den Belag"
    items:
      - "2 Dosen Mandarinen, gut abgetropft"
      - "etwas Puderzucker zum Bestäuben nach Wunsch"

steps:
  - title: "Teig kneten"
    description: "Verknete Mehl, Butter, Zucker, Ei, Salz und Backpulver rasch zu einem glatten Teig und drücke ihn in eine gefettete Springform mit kleinem Rand."
  - title: "Füllung rühren"
    description: "Rühre Quark, Schmand, Zucker, Eier, Vanillezucker, Puddingpulver, Öl und Zitronenschale zu einer cremigen Masse ohne Klümpchen."
  - title: "Mandarinen verteilen"
    description: "Streiche die Quarkmasse auf den Boden und verteile die gut abgetropften Mandarinen locker darauf, damit jede Ecke Frucht bekommt."
  - title: "Kuchen backen"
    description: "Backe den Kuchen, bis der Rand goldgelb aussieht und die Mitte nur noch leicht wackelt, dann nimm ihn heraus."
  - title: "Auskühlen lassen"
    description: "Lass den Kuchen vollständig auskühlen, damit die Füllung fest wird, und bestäube ihn erst dann nach Wunsch mit etwas Puderzucker."
