ingredients:
  - title: "Kikärtsgryta"
    items:
      - "2 burkar kikärtor (ca 800 g totalt)"
      - "1 burk kokosmjölk (400 ml)"
      - "1 st stor gul lök"
      - "2 st vitlöksklyftor"
      - "1 msk olja till stekning"
      - "1 msk gul curry"
      - "0.5 tsk salt"
      - "1 krm svartpeppar"
  - title: "Servering"
    items:
      - "4 portioner kokt ris"
      - "Färsk koriander eller persilja"

steps:
  - title: "Fräs löken"
    description: "Skala och finhacka den gula löken och vitlöken, fräs dem mjuka i olja i en rymlig kastrull eller stekpanna på medelvärme."
  - title: "Krydda basen"
    description: "Tillsätt curry och låt den fräsa med löken i ca 30 sekunder så att smakerna blommar ut ordentligt."
  - title: "Tillsätt kikärtor"
    description: "Häll av och skölj kikärtorna noga, vänd ner dem i kastrullen och rör om så de täcks av kryddorna."
  - title: "Häll på kokosmjölk"
    description: "Tillsätt kokosmjölken, rör om ordentligt och låt alltsammans koka upp under omrörning."
  - title: "Sjud grytan"
    description: "Låt grytan sjuda försiktigt i ca 10 minuter så att smakerna kokar ihop och såsen tjocknar något."
  - title: "Smaka av och servera"
    description: "Smaka av med salt och peppar; servera din snabba kikärtsgryta rykande het tillsammans med nykokt ris."
