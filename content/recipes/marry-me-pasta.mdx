ingredients:
  - title: "Für die Pasta"
    items:
      - "400 g Penne"
      - "Salz für das Kochwasser"
  - title: "Für die Sauce"
    items:
      - "1 kleine Zwiebel"
      - "3 Knoblauchzehen"
      - "120 g getrocknete Tomaten in Öl"
      - "2 EL Tomatenmark"
      - "1 TL Chiliflocken"
      - "250 ml Sahne"
      - "200 ml Gemüsebrühe"
      - "80 g frisch geriebener Parmesan"
      - "1 EL Olivenöl"
      - "schwarzer Pfeffer"
      - "1 Handvoll frischer Basilikum"

steps:
  - title: "Wasser aufsetzen"
    description: "Setze reichlich Salzwasser auf und koche die Penne, bis sie bissfest sind."
  - title: "Zutaten vorbereiten"
    description: "Schneide Zwiebel, Knoblauch und getrocknete Tomaten klein und reibe den Parmesan fein."
  - title: "Aromen anschwitzen"
    description: "Erhitze Olivenöl und brate Zwiebel, Knoblauch und Chiliflocken an, bis alles weich ist und duftet."
  - title: "Sauce aufbauen"
    description: "Rühre Tomatenmark und getrocknete Tomaten ein, lösche mit Brühe und Sahne ab und lasse die Sauce sanft köcheln, bis sie cremig aussieht."
  - title: "Käse einrühren"
    description: "Gib den Parmesan in die etwas ruhigere Sauce und rühre, bis sie glatt, glänzend und sämig wird."
  - title: "Alles verbinden"
    description: "Mische die Penne mit der Sauce, lockere bei Bedarf mit etwas Kochwasser und serviere mit Basilikum und extra Parmesan."
