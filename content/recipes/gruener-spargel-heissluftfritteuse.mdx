ingredients:
  - title: "Für den Spargel"
    items:
      - "500 g grüner Spargel"
      - "1 EL Olivenöl"
      - "1/2 TL Salz"
      - "1/4 TL schwarzer Pfeffer"
      - "1/2 TL Knoblauchpulver"
      - "1 TL Zitronenabrieb"
  - title: "Zum Servieren"
    items:
      - "1/2 Zitrone"
      - "2 EL frisch geriebener Parmesan"

steps:
  - title: "Spargel vorbereiten"
    description: "Wasche den Spargel, schneide die holzigen Enden ab und trockne die Stangen gut, damit sie später besser bräunen."
  - title: "Würzen und mischen"
    description: "Gib Spargel, Öl, Salz, Pfeffer, Knoblauchpulver und Zitronenabrieb in eine Schüssel und mische alles, bis die Stangen rundum leicht glänzen."
  - title: "In den Korb legen"
    description: "Lege den Spargel locker in den Korb der Heißluftfritteuse, damit die Luft gut zirkulieren kann und die Spitzen nicht dämpfen."
  - title: "Golden garen"
    description: "Gare den Spargel, bis die Spitzen leicht gebräunt sind, es fein duftet und die Stangen zart, aber noch bissfest wirken."
  - title: "Anrichten und beenden"
    description: "Richte den Spargel sofort an, gib etwas Zitronensaft und Parmesan darüber und serviere ihn noch warm."
